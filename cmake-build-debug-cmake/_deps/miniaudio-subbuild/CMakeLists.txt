cmake_minimum_required(VERSION 4.0.2)

# Reject any attempt to use a toolchain file. We must not use one because
# we could be downloading it here. If the CMAKE_TOOLCHAIN_FILE environment
# variable is set, the cache variable will have been initialized from it.
unset(CMAKE_TOOLCHAIN_FILE CACHE)
unset(ENV{CMAKE_TOOLCHAIN_FILE})

# We name the project and the target for the ExternalProject_Add() call
# to something that will highlight to the user what we are working on if
# something goes wrong and an error message is produced.

project(miniaudio-populate NONE)



include(ExternalProject)
ExternalProject_Add(miniaudio-populate
                     "UPDATE_DISCONNECTED" "False" "EXTERNALPROJECT_INTERNAL_ARGUMENT_SEPARATOR" "URL" "https://raw.githubusercontent.com/mackron/miniaudio/master/miniaudio.h" DOWNLOAD_NO_EXTRACT YES
                    SOURCE_DIR          "/Users/fredrikkarlsaune/Desktop/dev/bilsim/cmake-build-debug-cmake/_deps/miniaudio-src"
                    BINARY_DIR          "/Users/fredrikkarlsaune/Desktop/dev/bilsim/cmake-build-debug-cmake/_deps/miniaudio-build"
                    CONFIGURE_COMMAND   ""
                    BUILD_COMMAND       ""
                    INSTALL_COMMAND     ""
                    TEST_COMMAND        ""
                    USES_TERMINAL_DOWNLOAD  YES
                    USES_TERMINAL_UPDATE    YES
                    USES_TERMINAL_PATCH     YES
)


ExternalProject_Get_Property(miniaudio-populate DOWNLOADED_FILE)
get_filename_component(dlFileName "${DOWNLOADED_FILE}" NAME)

ExternalProject_Add_Step(miniaudio-populate copyfile
  COMMAND    "/Applications/CLion.app/Contents/bin/cmake/mac/aarch64/bin/cmake" -E copy_if_different
             "<DOWNLOADED_FILE>" "/Users/fredrikkarlsaune/Desktop/dev/bilsim/cmake-build-debug-cmake/_deps/miniaudio-src"
  DEPENDEES  patch
  DEPENDERS  configure
  BYPRODUCTS "/Users/fredrikkarlsaune/Desktop/dev/bilsim/cmake-build-debug-cmake/_deps/miniaudio-src/${dlFileName}"
  COMMENT    "Copying file to SOURCE_DIR"
)
